def initialize(file)
  __debug_handle(file, leader: "initialize | #{file.class}")
  @handle =
    case file
      when String     then File.new(file)
      when FileHandle then file.handle
      else                 file
    end
  raise "#{file.class} incompatible" unless self.class.compatible?(@handle)
end
