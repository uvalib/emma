def emit_page_favicon(**opt)
  @page_favicon ||= DEFAULT_PAGE_FAVICON
  opt = meta_options(**opt)
  favicon_link_tag(@page_favicon, opt)
end

def page_favicon
  if block_given?
    set_page_favicon(yield)
  else
    emit_page_favicon
  end
end
