def char(v)
  v.is_a?(Integer) ? v.chr(Encoding::UTF_8) : v.chr
end

def pad_char(v, pad: nil, left: nil, right: nil)
  left  ||= pad || THIN_SPACE
  right ||= pad || left
  [left, v, right].map { |c| char(c) }.join
end
