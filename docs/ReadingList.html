def has_artifact?(item)
  if item.is_a?(String)
    artifacts.any? { |a| a.id == item }
  else
    artifacts.include?(item)
  end
end

def has_title?(item)
  if item.is_a?(String)
    titles.any? { |t| t.bookshareId == item }
  else
    titles.include?(item)
  end
end
