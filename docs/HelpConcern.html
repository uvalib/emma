def get_help_entry(topic)
  entry = HELP_ENTRY[topic]&.except(:content_html)
  if entry
    entry[:content] ||= 'TODO' # TODO: JSON/XML help content
  else
    entry = { error: 'Not a help topic.' }
  end
  entry
end
