def initialize(app, opt = nil)
  opt = DEFAULT_OPTIONS.deep_merge(opt || {})
  @store_options = opt[:store_options] || {}
  @expires_in    = opt[:expires_in] || @store_options[:expires_in]
  @store_options[:expires_in] = @expires_in
  super(app, opt)
end
