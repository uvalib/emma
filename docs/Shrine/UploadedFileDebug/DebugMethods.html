def __debug_upload(*args, **opt, &block)
  meth = args.shift
  meth = meth.to_s.upcase if meth.is_a?(Symbol)
  opt[:leader] = ':::SHRINE::: UploadedFile'
  opt[:separator] ||= ' | '
  __debug_items(meth, *args, opt, &block)
end
