def wf_set_records_state(*items, state: nil)
  items = succeeded if items.blank?
  items.each do |item|
    next unless item.is_a?(Upload)
    state ||= current_state
    item.set_phase(workflow_phase)
    item.set_state(state, workflow_column)
  end
end
