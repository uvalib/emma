def get_token(code, params = {}, opts = {})
  token_params = {
    grant_type: 'authorization_code',
    code:       code,
    client_id:  @client.id,
  }
  token_params.merge!(@client.redirection_params)
  token_params.merge!(params)
  token_params.stringify_keys!
  @client.get_token(token_params, opts)
end
